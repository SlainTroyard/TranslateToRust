# Rustify 环境变量配置示例
# 复制此文件为 .env 并填入你的 API Key
#
# 说明：
# - .env 文件用于存放敏感信息（API Key），不应提交到 Git
# - rustify.toml 用于存放其他配置，可以提交到 Git
# - 两者会自动合并，.env 优先级更高

# ============ LLM 配置（主翻译模型） ============
RUSTIFY_LLM_API_KEY=your-api-key-here

# 以下配置建议放在 rustify.toml 中，这里仅作参考
# RUSTIFY_LLM_MODEL=deepseek-chat
# RUSTIFY_LLM_BASE_URL=https://api.deepseek.com
# RUSTIFY_LLM_TEMPERATURE=1.0
# RUSTIFY_LLM_TOP_P=1.0
# RUSTIFY_LLM_MAX_TOKENS=8192

# ============ Analyzer 配置（深度分析模型） ============
RUSTIFY_ANALYZER_API_KEY=your-api-key-here

# RUSTIFY_ANALYZER_MODEL=deepseek-reasoner
# RUSTIFY_ANALYZER_BASE_URL=https://api.deepseek.com
